# CUDA runtime libraries — required by CTranslate2 (for Whisper GPU) and by
# llama-cpp-python's CUDA build (for LLM GPU). NVIDIA publishes them as Python
# wheels that drop DLLs into <venv>/Lib/site-packages/nvidia/<lib>/bin/. Our
# sidecar's __main__.py calls os.add_dll_directory() on those paths before any
# CUDA-linked import so the OS loader finds cublas64_12.dll, cudnn_*.dll, and
# cudart64_12.dll. cudnn pinned to 9.* per faster-whisper's install docs;
# cuda-runtime pulls in cudart64_12.dll which the cu12x llama-cpp-python wheel
# also needs. All sys_platform == 'win32' — mac/Linux installs stay slim.
nvidia-cublas-cu12; sys_platform == 'win32'
nvidia-cudnn-cu12==9.*; sys_platform == 'win32'
nvidia-cuda-runtime-cu12; sys_platform == 'win32'
